Deklarativ konfigurasjonsstyring

  • Deklarativ fremfor imperativ
  • Gjenskape et miljø fra forskjellige utgangspunkt 
apt-get install vim

echo "set nocp" >> /etc/vim/vimrc
yum install vim

echo "set nocp" >> /etc/vim/vimrc
package { "vim":
  ensure => present,
}

file { "/etc/vim/vimrc":
  content => "set nocp",
}
